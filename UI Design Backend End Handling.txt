6. API Example (Create User)
[HttpPost]
public async Task<IActionResult> CreateUser(UserRequest request)
{
    // Step 1
    // User table insert

    var userSql = @"
    INSERT INTO [User] (Name)
    VALUES (@Name);

    SELECT CAST(SCOPE_IDENTITY() as int);
    ";

    int userId = await _db.ExecuteScalarAsync<int>(userSql, new
    {
        request.Name
    });

    // Step 2
    // Interests save in junction table

    foreach (var interestId in request.Interests)
    {
        var sql = @"INSERT INTO UserInterest (UserId, InterestId)
                    VALUES (@UserId, @InterestId)";

        await _db.ExecuteAsync(sql, new
        {
            UserId = userId,
            InterestId = interestId
        });
    }

    return Ok(userId);
}

C# API
[HttpPost]
public async Task<IActionResult> CreateOrder(OrderRequest request)
{
    var orderSql = @"
    INSERT INTO [Order] (CustomerName,OrderDate)
    VALUES (@CustomerName,GETDATE());

    SELECT CAST(SCOPE_IDENTITY() as int)";

    int orderId = await _db.ExecuteScalarAsync<int>(orderSql,new
    {
        request.CustomerName
    });

    foreach(var item in request.Items)
    {
        var sql = @"INSERT INTO OrderItem
                    (OrderId,ProductName,Quantity,Price)
                    VALUES (@OrderId,@ProductName,@Quantity,@Price)";

        await _db.ExecuteAsync(sql,new
        {
            OrderId = orderId,
            item.ProductName,
            item.Quantity,
            item.Price
        });
    }

    return Ok(orderId);
}

[HttpPost("upload")]
public async Task<IActionResult> UploadFile(IFormFile file,int employeeId)
{
    var fileName = file.FileName;
    var path = "uploads/" + fileName;

    using(var stream = new FileStream(path,FileMode.Create))
    {
        await file.CopyToAsync(stream);
    }

    var sql = @"INSERT INTO EmployeeDocument
                (EmployeeId,FileName,FilePath,UploadDate)
                VALUES (@EmployeeId,@FileName,@FilePath,GETDATE())";

    await _db.ExecuteAsync(sql,new
    {
        EmployeeId = employeeId,
        FileName = fileName,
        FilePath = path
    });

    return Ok();
}
