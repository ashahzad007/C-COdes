using System.Diagnostics;
using System.Runtime.CompilerServices;

namespace demo
{
    public delegate void EventDelegate (object sender, EventAnnouncer e);
    // data class container
    public class EventAnnouncer : EventArgs
    {
        public string Title { get; set; }
        public int Discount { get; set; }

        public EventAnnouncer(string title, int discount)
        {
            this.Discount = discount;
            this.Title = title;
        }
    }
    // publisher + event raiser / invoker
    public class Shop
    {
       public event EventDelegate Deli;
        
        public void SaleAnnounce(EventAnnouncer eventAnnouncer)
        {
           
            Console.WriteLine(eventAnnouncer.Title);
            Console.WriteLine(eventAnnouncer.Discount);
            Console.WriteLine("sale is on ");
            Deli?.Invoke(this,eventAnnouncer);
        }
        
        
    }
    

    public class SMSService
    {
        public void sms(object sender , EventAnnouncer eventAnnouncer)
        {
            Console.WriteLine("SMS Sent");
        }
    }

    public class EmailService
    {
        public void email(object sender , EventAnnouncer eventAnnouncer) { Console.WriteLine("email sent"); }
    }
    class Program
    {
        static void Main(string[] args)
        {

            EventAnnouncer EventAnnouncer = new EventAnnouncer("Disounct Sale", 40);
            SMSService SMSService = new SMSService();
            EmailService EmailService = new EmailService();
            Shop shop = new Shop();
            shop.Deli += SMSService.sms;
            shop.Deli += EmailService.email;
            shop.SaleAnnounce(EventAnnouncer);
            Console.ReadKey();


        }
    }


}

        {
            Title = title;
            Discount = discount;
        }

    public class Shop
    {
        // now you have two options use built in event or declare delagte in name space

        // please note event will invoke in method

        {
            Console.WriteLine($"{eventAnnouncer.Title} and {eventAnnouncer.Discount}");
            AnnounceSale.Invoke( sender , eventAnnouncer );
            //AnnounceSale(sender, eventAnnouncer); both ways you can use invoke or this way
        }
    }

    // step 4 , create subscriber classes and their methods

Rule: Step-by-step.
Har step ka reason + change clear.

🧱 STEP 0 — Current Situation (baseline)

Abhi:

Order → NotificationService.SendSMS()
Order → NotificationService.SendEmail()

Problem:

❌ Tight coupling
❌ Order ko pata hai Email + SMS
❌ Future change = Order modify

Goal:

✅ Order sirf announce kare → “OrderPlaced”

🧱 STEP 1 — Delegate introduce karte hain
🎯 Reason

Order ko yeh nahi pata hona chahiye:

Email kaun bhej raha

SMS kaun bhej raha

Order sirf bole:

👉 “Agar koi interested hai → call ho jao”

Yahan delegate ka entry point.

✅ Change #1 — Delegate type

Program ke andar define karo:

delegate void OrderPlacedHandler(string message);

Meaning:

👉 “Jo bhi handler hoga, string lega”

✅ Change #2 — OrderService publish karega

OrderService:

public OrderPlacedHandler OnOrderPlaced;

Aur:

OnOrderPlaced?.Invoke("Order Confirmed");

Meaning:

👉 “Agar koi subscribed hai → callback call”

🧠 Yahan kya hua?

Delegate = function slot
Invoke = callback execution

🧱 STEP 2 — NotificationService subscriber banega
🎯 Reason

Ab NotificationService ko OrderService call nahi karega.

🧠 What is Func<>?

Func<> is a generic built-in delegate.

It represents a method that:

✔ Returns a value
✔ Can take 0…16 parameters

Example:

Func<int, int, int> add = (a, b) => a + b;

Meaning:

Func<T1, T2, TResult>

⏳ Before Func<> existed…

Developers had to create manual custom delegates every time.

Example 👇

❌ WITHOUT Func (Old Style)
using System;

{
