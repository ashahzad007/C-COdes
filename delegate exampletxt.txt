using System.Diagnostics;
using System.Runtime.CompilerServices;

namespace demo
{
    public delegate void EventDelegate (object sender, EventAnnouncer e);
    // data class container
    public class EventAnnouncer : EventArgs
    {
        public string Title { get; set; }
        public int Discount { get; set; }

        public EventAnnouncer(string title, int discount)
        {
            this.Discount = discount;
            this.Title = title;
        }
    }
    // publisher + event raiser / invoker
    public class Shop
    {
       public event EventDelegate Deli;
        
        public void SaleAnnounce(EventAnnouncer eventAnnouncer)
        {
           
            Console.WriteLine(eventAnnouncer.Title);
            Console.WriteLine(eventAnnouncer.Discount);
            Console.WriteLine("sale is on ");
            Deli?.Invoke(this,eventAnnouncer);
        }
        
        
    }
    

    public class SMSService
    {
        public void sms(object sender , EventAnnouncer eventAnnouncer)
        {
            Console.WriteLine("SMS Sent");
        }
    }

    public class EmailService
    {
        public void email(object sender , EventAnnouncer eventAnnouncer) { Console.WriteLine("email sent"); }
    }
    class Program
    {
        static void Main(string[] args)
        {

            EventAnnouncer EventAnnouncer = new EventAnnouncer("Disounct Sale", 40);
            SMSService SMSService = new SMSService();
            EmailService EmailService = new EmailService();
            Shop shop = new Shop();
            shop.Deli += SMSService.sms;
            shop.Deli += EmailService.email;
            shop.SaleAnnounce(EventAnnouncer);
            Console.ReadKey();


        }
    }


}
======================================
step 1 create delegate  public delegate void EventDelegate (object sender, EventAnnouncer e);
step 2 create publisher class who publish , 
step 3 in publisher class you create public event EventDelegate Deli;
step 4  Deli?.Invoke(this,eventAnnouncer);
step 5 , create subscribers methods which match the delegate signature
step 6  subscribe shop.Deli += SMSService.sms; shop.Deli += EmailService.email;
step 7 fire event shop.SaleAnnounce(EventAnnouncer);

using System.Runtime.CompilerServices;

namespace EventDemo
{
    // step 1 create event rasier class for data collection
    // delegate creation if you dont want to use builtin delegete
    //public delegate void EventHandlerCustom(Object sender, EventAnnoucer eventAnnouncer);
    public class EventAnnoucer : EventArgs
    {
        public string Title { get; set; }
        public string Discount { get; set; }

        public EventAnnoucer(string title, string discount)
        {
            Title = title;
            Discount = discount;
        }
    }

    // step 2 Publisher Class and Event Invoker  
    public class Shop
    {
        // now you have two options use built in event or declare delagte in name space 
        // step 2 
        public event EventHandler<EventAnnoucer> AnnounceSale; // this is built in no need to declare delegate
        // please note event will invoke in method 
        // or if you want to use your own custome delegate 
        //public event EventHandlerCustom CustomAnnounceSale;

        // step 3 create method for event invoke
        public void AnnounceSaleMethod(Object sender , EventAnnoucer eventAnnouncer)
        {
            Console.WriteLine($"{eventAnnouncer.Title} and {eventAnnouncer.Discount}");
            AnnounceSale.Invoke( sender , eventAnnouncer );
            //AnnounceSale(sender, eventAnnouncer); both ways you can use invoke or this way
        }
    }

    // step 4 , create subscriber classes and their methods 
    public class SendEmailService
    {
        public void SendEmail(Object sender, EventAnnoucer e)
        {
            Console.WriteLine("Email Sent");
        }    
    }

    public class SendSmsService 
    {
        public void SendSMS(Object sender, EventAnnoucer e)
        {
            Console.WriteLine("SMS Sent");
        }
    }


    class Program
    {
        static void Main(string[] args)
        {
            // step 5 calling  and subscribing
            // please note you can subscribe first and then fire the event
            EventAnnoucer e = new EventAnnoucer("Our Sale is Started","30% discount");
            SendEmailService emailService = new SendEmailService();
            SendSmsService smsService = new SendSmsService();
            // please note you have to subscribe first other wise your event will be null

            Shop shop = new Shop();
            shop.AnnounceSale += emailService.SendEmail;
            shop.AnnounceSale += smsService.SendSMS;

            // step 6 , this method has event Invoke so in last you run this method as signal
            // that sale is strated
            shop.AnnounceSaleMethod(shop,e);
            
        }
    }
}
